    -------------------------------------------------------------
    1★ == 10% * 2 ; 5★ == 100% ; - (letter-spacing/2) 间距差值
    循环出 .st-1 至 .st-10 名字; 
    .st-1{width:calc(10% - (letter-spacing/2));}
    .st-10{width:calc(100% - (letter-spacing/2));}
    -------------------------------------------------------------

    // 默认变量

        $bg-color:  $color,     //背景色
        $spacing:   4px,        //间距
        $bg-icon:   $solid,     //背景样式
        $icon:      $hollow     //前景样式   
    ){
      ...
    }


    // sass
    --------------------

    @include stars($size, $color, $bg-color, $spacing, $bg-icon, $icon){

    };
